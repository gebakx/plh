from nltk import word_tokenize, pos_tag, ne_chunk
 
sentence = "Mark Pedersen and John Smith are working at Google " + \
           "since 1994 for $1000 per week."

x = pos_tag(word_tokenize(sentence))
res = ne_chunk(x)
print(res)
